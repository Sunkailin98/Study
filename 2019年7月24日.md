    //定义方法完成成绩排序并输出前三名的功能

    public void showTop3(int[] scores){
        Arrays.sort(scores);
        int num = 0;
        for(int i = scores.length - 1;i >= 0;i--){
            if(scores[i] < 0 || scores[i] > 100){
                continue;
            }
            num++;
            if(num > 3){
                break;
            }
            System.out.println(scores[i]);
        }
    }
